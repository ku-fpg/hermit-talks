\section{Commands}

\begin{frame}
\frametitle{HERMIT Commands}

\vspace{-1ex}

\begin{itemize}
\item
Core-specific rewrites, e.g.
\begin{itemize}
\item beta-reduce
\item eta-expand '$x$
\item case-split '$x$
\item inline
\end{itemize}

\item
Strategic traversal combinators (from KURE), e.g.
\begin{itemize}
\item any-td $r$
\item repeat $r$
\item innermost $r$
\end{itemize}


\item
Navigation, e.g.
\begin{itemize}
\item up, down, left, right, top
\item binding-of '|foo|
\item app-fun, app-arg, let-body, \ldots
\end{itemize}

\item
Version control, e.g.
\begin{itemize}
\item log
\item back
\item step
\item save ``myscript.hss''
\end{itemize}

\end{itemize}

\end{frame}
